  1. x2c("4c") --> '\x4c'，这里的0xdf靓了

    char TC_Common::x2c(const string &sWhat)
{
 register char digit;

 if(sWhat.length() < 2)
 {
     return '\0';
 }

 digit = (sWhat[0] >= 'A' ? ((sWhat[0] & 0xdf) - 'A')+10 : (sWhat[0] - '0'));
 digit *= 16;
 digit += (sWhat[1] >= 'A' ? ((sWhat[1] & 0xdf) - 'A')+10 : (sWhat[1] - '0'));

 return(digit);
}

  9. 这样的一段代码是为了占用fd 0, 1, 2，免得打开的文件、管道或是socket用了0，1，2，受程序里的printf等影响

    int fd = open("/dev/null", O_RDWR );
if(fd != -1)
{
dup2(fd, 0);
dup2(fd, 1);
dup2(fd, 2);
}
else
{
close(0);
close(1);
close(2);
}
